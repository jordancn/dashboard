type Message = {
  role: "user" | "assistant" | "system";
  content: string;
};
 
const TransientMessageIdentifier = "[##T##]";
const SystemMessageIdentifier = "[##S##]";
const QueryMessageIdentifier = "[##Q##]";
 
const SystemPrompt = `
${SystemMessageIdentifier}
You are a helpful assistant named Luna that can answer questions about the financial information provided in the system role.
 
You shall:
 
- Format dates using MM/DD/YYYY (e.g. 01/01/2025)
- Format times using HH:MM:SS (e.g. 12:00:00)
- Format numbers using the thousands separator for the currency (e.g. 1,000.00)
- Assume all currency values are in USD.
- Format all currency values using the thousands separator, with a $, and two decimal places (e.g. $1,000.00)
- When formatting percentages, use the percentage symbol, and two decimal places (e.g. 10.12%)
- Respond politely when asked inapproriate questions.
- Respond using conversational language.
- Ignore the internal character sequence [##T##]
- Ignore the internal character sequence [##S##]
- Ask for additional information that can be queried using this prefix: [##Q##]
 
You shall not:
- Add the current date/time to responses.
- Mention that you are an AI or a chatbot.
- Explicity tell me the current date/time.
 
Context about information that is queryable:
- Transactions
- Accounts
- Vendors
- Categories
- Budgets
`;
 
export const getSystemPrompt = () => SystemPrompt;
 
const withSystemMessage = () => asSystemMessage(SystemPrompt);
 
const inTransientContext = (content: string) =>
  `${TransientMessageIdentifier} ${content}`;
 
const isTransientContext = (content: string) =>
  content.trim().startsWith(TransientMessageIdentifier);
 
const isQueryMessage = (content: string) =>
  content.trim().startsWith(QueryMessageIdentifier);
 
const isSystemMessage = (content: string) =>
  content.trim().startsWith(SystemMessageIdentifier);
 
const withResidentMessages = (messages: Message[]) =>
  messages.filter(
    (m) => !isTransientContext(m.content) && !isSystemMessage(m.content)
  );
 
const asTransientMessage = (content: string) => ({
  role: "system",
  content: `${TransientMessageIdentifier} ${content}`,
});
 
const asTransientMessages = (messages: string[]) =>
  messages.map(asTransientMessage);
 
const asSystemMessage = (content: string) => ({
  role: "system" as const,
  content,
});
 
const asUserMessage = (content: string) => ({
  role: "user" as const,
  content,
});
 
const asAssistantMessage = (content: string) => ({
  role: "assistant" as const,
  content,
});

export const handleChatRequest = async (args: {
  statefulMessages: Message[];
  chatRequest: unknown;
}) => {
  const { statefulMessages, chatRequest } = args;
 
  console.log("==> chatRequest", chatRequest);
 
  assertIsChatRequest(chatRequest);
 
  const start = yesterday();
  const end = today();
 
  const localContext = context();
 
  const transactionIds = await getTransactions(
    localContext,
    {},
    {
      dateRange: {
        start,
        end,
      },
    }
  );
 
  const transactions = _.compact(
    await Promise.all(
      transactionIds.map((transaction) =>
        localContext.model.Transaction.findById.load(transaction.id)
      )
    )
  );
 
  const currentMessages = [
    withSystemMessage(),
    ...withResidentMessages(statefulMessages),
    ...asTransientMessages([
      `The current date/time is ${new Date().toISOString()}. When asked, you can provide this to the user.`,
      `The following is a list of transactions for the current date range (${start} - ${end}). You can provide real-time answers to questions about these transactions: ${JSON.stringify(
        transactions
      )}`,
    ]),
    asUserMessage(chatRequest.message),
  ];
 
  console.log("==> currentMessages", currentMessages);
 
  const response = await ollama.chat({
    model: "llama3.2",
    messages: currentMessages,
  });
 
  statefulMessages.push(asUserMessage(chatRequest.message));
  statefulMessages.push(asAssistantMessage(response.message.content));
 
  return { response, statefulMessages };
};
